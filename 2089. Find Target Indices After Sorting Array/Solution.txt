class Solution {
    public List<Integer> targetIndices(int[] nums, int target) {
        int left=0;
        int right=0;
        for(int i=0; i<nums.length;i++){
            if(nums[i]<target){
                left++;
            }
            else if(nums[i]==target){
                right++;
            }
        }
        List<Integer> ans = new ArrayList<>();
            
        for(int i=0; i<right; i++){
            ans.add(left+i);
        }
        return ans;
    }
}
