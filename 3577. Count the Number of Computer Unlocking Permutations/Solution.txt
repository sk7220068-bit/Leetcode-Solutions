class Solution {
    public int countPermutations(int[] complexity) {
        int MOD=1000000007;
        for(int i=1; i<complexity.length; i++){
            if(complexity[i]<=complexity[0]){
                return 0;
            }
        }

        long ans =1;
        for(int i=1; i<=(complexity.length-1); i++){
            ans=(ans*i)%MOD;
        }
        return (int)ans;

    }
}
