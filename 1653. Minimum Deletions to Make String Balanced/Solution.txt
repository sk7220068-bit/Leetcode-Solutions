class Solution {
    public int minimumDeletions(String s) {
        int bcount=0; 
        int deletion=0;

        for(char ch : s.toCharArray()){
            if(ch=='b'){
                bcount++;
            }
            else if(bcount>0){
                bcount--;
                deletion++;
            }
        } 
        return deletion;

    }
}
