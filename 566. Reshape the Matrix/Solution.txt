class Solution {
    public int[][] matrixReshape(int[][] mat, int r, int c) {
        int n=mat.length;
        int m=mat[0].length;
        if((m*n)!=(r*c)){
            return mat;
        }

        int[][]ans = new int[r][c];
        int idx=0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                ans[idx/c][idx%c]=mat[i][j];
                idx++;
            }
        }
        return ans;
    }
}
