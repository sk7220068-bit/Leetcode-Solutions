class Solution {
    public int matrixSum(int[][] nums) {
        for(int i=0; i<nums.length; i++){
            Arrays.sort(nums[i]);
        }
        int score=0;

        for(int i=0; i<nums[0].length; i++){
            int max=nums[0][i];
            for(int j=1; j<nums.length; j++){
                if(max<nums[j][i]){
                    max=nums[j][i];
                }
            }
            score+=max;
        }
        return score;
    }
}
