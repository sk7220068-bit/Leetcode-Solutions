class Solution {
    public int bestClosingTime(String customers) {
        int profit =0; 
        int maxProfit = 0;
        int ans = 0;
        for(int i=0; i<customers.length();i++){
            int c=customers.charAt(i);

            if(c=='Y'){
                profit++;
            }else{
                profit--;
            }

            if(profit>maxProfit){
                maxProfit=profit;
                ans=i+1;
            }
        }
        return ans;
    }
}
