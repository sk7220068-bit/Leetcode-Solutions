class Solution {
    public int alternateDigitSum(int n) {
        int sum = 0;
        int rev=0;
        int sign = 1;
        while(n>0){
            rev = rev *10 + n % 10;
            n /= 10;
        }
        while(rev>0){
            sum = sum + (rev%10 * sign);
            rev /= 10;
            sign *= -1;
        }
        return sum;
    }

}
