class Solution {
    public int minimumDifference(int[] nums, int k) {
        Arrays.sort(nums);
        if(k==1){
            return 0;
        }

        int i=0; 
        int j=k-1;
        int min= Integer.MAX_VALUE;
        while(j<nums.length){
            min=Math.min(min, nums[j++] - nums[i++]);
        }
        return min;
    }
}
