import java.util.*;
class Solution {
    public List<Integer> findDisappearedNumbers(int[] nums) {
        
        Set<Integer> result = new HashSet<>();

        for(int i=0; i<nums.length; i++){
            result.add(nums[i]);
        }

        List<Integer> ans = new ArrayList<>();

        for(int i=1; i<=nums.length; i++){
            if(!result.contains(i)){
                ans.add(i);
            }
        }
        return ans;
        
    }
}
