class Solution {
    public int minOperations(String s) {
        int mis = 0;
        int n = s.length();
        for (int i = 0; i < n; i++) {
            char ex = (i % 2 == 0) ? '0' : '1';

            if (s.charAt(i) != ex) {
                mis++;
            }
        }

        return Math.min(mis, n - mis);
    }
}
