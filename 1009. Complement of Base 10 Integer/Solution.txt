class Solution {
    public int bitwiseComplement(int n) {
        if(n==0){
            return 1;
        }
        int count=0;
        int res=0;
        while(n>0){
            int r=n%2;
            if(r==0){
                res+=(1<<count);
            }
            count++;
            n/=2;
        }
        return res;
    }
}
